mask = np.logical_and(batting.loc[:,'teamID'] == 'SLN',batting.loc[:,'yearID'] >= 1900 )
stl_batting = batting.loc[mask,['yearID','H','AB','HR']].groupby('yearID').sum()
stl_batting.loc[:,'BA'] = stl_batting.loc[:,'H'] / stl_batting.loc[:,'AB']

maskchi = np.logical_and(batting.loc[:,'teamID'] == 'CHN',batting.loc[:,'yearID'] >= 1900 )
chi_batting = batting.loc[maskchi,['yearID','H','AB','HR']].groupby('yearID').sum()
chi_batting.loc[:,'BA'] = chi_batting.loc[:,'H'] / chi_batting.loc[:,'AB']
